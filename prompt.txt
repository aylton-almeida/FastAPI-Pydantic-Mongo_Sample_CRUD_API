    name: str
    age: PositiveInt
    email: EmailStr

    @model_validator(mode="before")
    @classmethod
    def _set_is_adult(cls, data) -> Person:
        """Calculates if person is an adult based on age"""

        if (age := data.get("age", 0)) and age >= 18:
            data["is_adult"] = True

        return data
